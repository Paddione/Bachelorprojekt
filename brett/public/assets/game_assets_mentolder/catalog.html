Three.js · Material Library brett/public/assets/materials.js neu · single source
  • inkBody — MeshStandardMaterial({ color: 0x17202e, metalness: 0.55, roughness: 0.65 }) — Waffen-Bodies, Fässer
  • brassDetail — emissive Brass für Trigger, Sights, Klingen
  • woodWarm — Keulen, Holzkisten, Tsuka
  • concrete — Pfeiler, Mauer, Boden-Variation
  • edgeBrass — LineBasicMaterial({ color: 0xd7b06a, opacity: 0.35 }) als EdgesGeometry-Wrapper
  • Zentrale applySignature(mesh)-Funktion fügt automatisch Edges-Brass an alle neuen Meshes
import * as THREE from 'three';

export const inkBody = new THREE.MeshStandardMaterial({
  color: 0x17202e, metalness: 0.55, roughness: 0.65,
});
export const brassDetail = new THREE.MeshStandardMaterial({
  color: 0xd7b06a, metalness: 0.85, roughness: 0.35,
  emissive: 0x4a3814, emissiveIntensity: 0.15,
});
export const woodWarm  = new THREE.MeshStandardMaterial({ color: 0x6a4a28, roughness: 0.85 });
export const concrete  = new THREE.MeshStandardMaterial({ color: 0x5a5852, roughness: 0.95 });
export const edgeBrass = new THREE.LineBasicMaterial({
  color: 0xd7b06a, transparent: true, opacity: 0.35,
});

// brass wireframe signature on every mesh that opts in
export function applySignature(mesh) {
  const edges = new THREE.EdgesGeometry(mesh.geometry, 25);
  mesh.add(new THREE.LineSegments(edges, edgeBrass));
  return mesh;
}

Beleuchtung & Boden

  • Aktuelle helle Szene-Beleuchtung → 30% reduziert, kühler Tone
  • HemisphereLight — sky = stille-blau (low-intensity), ground = ink-900
  • 1–2 SpotLights mit warmem Brass-Akzent über Spawn-Punkten
  • Brett-Bodenmaterial: concrete-Token + leichte Vignette
  • Coaching-Mode: gleiche Beleuchtung, nur weniger dramatisch (Intensität +20%)
// drop-in lighting rig — replace current AmbientLight setup
scene.add(new THREE.HemisphereLight(
  0x6fa8d8, // sky · stille-blau
  0x0b111c, // ground · ink-900
  0.45,
));
const spawn = new THREE.SpotLight(0xd7b06a, 1.2, 12, Math.PI/5, 0.4);
spawn.position.set(0, 6, 2);
scene.add(spawn);

// coaching-mode override
if (mode === 'coaching') hemi.intensity *= 1.2;
