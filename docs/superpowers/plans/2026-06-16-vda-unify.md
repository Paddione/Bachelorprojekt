# VDA-Unify — Unified View-Decision-Action Interface (Phase 1: Core Library + Subcommand-Skeletons)

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Single entry point `scripts/vda.sh` with shared helper library `scripts/lib/vda-core.sh` replaces 11+ inconsistent interactive scripts. Phase 1 establishes the VDA core library, extracts shared code from `ticket.sh` and `feature-promote.sh` into reusable modules, refactors both to thin dispatchers, creates skeleton subcommands for ticket CRUD/pipeline, and adds deprecation wrappers for legacy scripts. All new subcommand files stay under 150 lines; pure bash without fzf/gum/jq.

**Architecture:** `vda.sh` → case dispatch sourcing `lib/vda-core.sh` for View-Decision-Action helpers. Ticket subcommands live in `scripts/vda/ticket/*.sh` sourcing `_ticket-core.sh`. Promote phases extracted to `lib/promote-phases.sh`. Legacy scripts become thin deprecation wrappers (`echo → exec vda.sh`). `VDA_NONINTERACTIVE=1` enables deterministic CI/Factory mode with no prompts.

**Tech Stack:** Bash (`set -euo pipefail`), PostgreSQL via `kubectl exec … psql`, go-task, BATS.

```bash
vda_choose() {
  local prompt="$1" opt; shift
  if [[ "${VDA_NONINTERACTIVE:-0}" = "1" || ! -t 0 ]]; then
    [[ $# -ge 1 ]] && echo "${1}" || return 1
    return 0
  fi
  select opt in "$@"; do
    [[ -n "$opt" ]] && { echo "$opt"; return 0; }
    echo "Invalid selection." >&2
  done
}

vda_confirm() {
  local prompt="${1:-Continue?}" default="${2:-y}"
  if [[ "${VDA_NONINTERACTIVE:-0}" = "1" || ! -t 0 ]]; then
    return 0
  fi
  local ans
  read -r -p "${prompt} [${default}] " ans || true
  ans="${ans:-$default}"
  [[ "$ans" =~ ^[yYjJ] ]]
}

vda_input() {
  local prompt="$1" default="${2:-}"
  if [[ "${VDA_NONINTERACTIVE:-0}" = "1" || ! -t 0 ]]; then
    [[ -n "$default" ]] && echo "$default" || return 1
    return 0
  fi
  local ans
  local p="${prompt}"
  [[ -n "$default" ]] && p="${p} [${default}]"
  read -r -p "${p}: " ans || true
  echo "${ans:-$default}"
}
```

```bash
  local json="{"
  local first=1 pair k v

    k="${pair%%=*}"
    v="${pair#*=}"
    # Shell-escape for JSON
    v="$(printf '%s' "$v" | sed 's/"/\\"/g')"

    first=0

  if command -v jq &>/dev/null; then

  else

}
```

```bash
#!/usr/bin/env bash
# scripts/vda.sh — Unified View-Decision-Action Interface
set -euo pipefail

SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
source "${SCRIPT_DIR}/lib/vda-core.sh"

show_help() {
  vda_header "vda.sh — VDA Unified Interface"
  echo ""
  echo "Usage: vda.sh <command> [options]"
  echo ""
  echo "Commands:"
  echo "  oracle [--goal <text>]     Task-Orakel (replaces task-oracle.sh)"
  echo "  promote [service]          Feature-Promotion (replaces feature-promote.sh)"
  echo "  frontmatter <file>         Frontmatter-Hook (replaces plan-frontmatter-hook.sh)"
  echo "  backup <sub> [args]        Backup/Recovery (delegates to backup-restore.sh)"
  echo "  ticket <sub> [args]        Ticket-CRUD + Pipeline (replaces ticket.sh)"
  echo "  factory-prep [--dry-run]   Factory PREP (replaces dispatcher-prep.sh)"
  echo "  brainstorm <sub> [args]    Brainstorm-Bridge (replaces brainstorm-bridge.sh)"
  echo "  help                       Show this help"
  echo "  version                    Show version"
}

show_version() { echo "vda.sh 1.0.0"; }

main() {
  [[ "${1:-}" = "--help" || "${1:-}" = "-h" ]] && { show_help; exit 0; }

  case "${1:-}" in
    oracle)
      shift
      exec "${SCRIPT_DIR}/vda/oracle.sh" "$@"
      ;;
    promote)
      shift
      exec "${SCRIPT_DIR}/vda/promote.sh" "$@"
      ;;
    frontmatter)
      shift
      exec "${SCRIPT_DIR}/vda/frontmatter.sh" "$@"
      ;;
    backup)
      shift
      exec "${SCRIPT_DIR}/vda/backup.sh" "$@"
      ;;
    ticket)
      shift
      exec "${SCRIPT_DIR}/vda/ticket.sh" "$@"
      ;;
    factory-prep)
      shift
      exec "${SCRIPT_DIR}/vda/factory-prep.sh" "$@"
      ;;
    brainstorm)
      shift
      exec "${SCRIPT_DIR}/vda/brainstorm.sh" "$@"
      ;;
    help|--help|-h)
      show_help
      ;;
    version|--version|-v)
      show_version
      ;;
    *)
      vda_error "Unknown command: ${1:-}. Use 'vda.sh help' for usage."
      exit 2
      ;;
  esac
}

main "$@"
```

```bash
# scripts/vda/ticket/_ticket-core.sh
# Shared PG helpers for ticket subcommands. Sourced by ticket.sh and vda/ticket/*.sh.
# Expects: NS, CTX, DB from sourcing context.

_pgpod() {
  kubectl --context="${CTX}" get pod -n "${NS}" -l "app=shared-db" -o jsonpath='{.items[0].metadata.name}' 2>/dev/null
```
